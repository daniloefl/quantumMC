117 void SchroedingerDiffusionMC::step(int n) {
118  uniform_real_distribution<double> rFlat(0, 1);
119  // shift walker
120  // psi = prod dx_j [ prod W P(x_n, x_{n-1}) ] psi(0)
121  // Weight function W = exp(-dt*(V - E))
122 
123  // P = exp( - m (x_n - x_{n-1})^2/dt ) incorporates the kinetic energy
124  // and it is simulated by the random displacement below
125  normal_distribution<double> rGaus(0.0, 1.0);
126  m_x[n][0] += sqrt(m_dt)*rGaus(rEngine); // shifts walker by a gaussian with width sqrt(delta t): diffusion!
127 
128  // now we need to incorporate W
129  // could be done weighting the psi function
130  // but it is inefficient
131  // so we kill walkers depending on W
132  // change in effective potential energy V - E causes the walker to die with certain probability
133  // https://www.thphys.uni-heidelberg.de/~wetzel/qmc2006/KOSZ96.pdf
134  // page 5
135  // probability of a surviving walker is proportional to W, but W is not bounded in principle
136  // so make it integer rounding it up or down depending on its fractional part
137  double W = exp(-m_dt*(V(m_x[n]) - m_E0));
138  int survivors = int(W);
139  if (W - survivors > rFlat(rEngine)) {
140  survivors++;
141  }
142  // survivors = W with probability frac(W) = W - int(W)
143  // and survivors = W+1 with probability 1 - frac(W)
144  // now we have an integer number of new particles
145  // make a new walker
146  if (survivors-1 > 0) {
147  m_x.resize(m_N+(survivors-1));
148  m_alive.resize(m_N+(survivors-1));
149  for (int i : irange<int>(0, survivors - 1)) {
150  m_x[m_N][0] = m_x[n][0];
151 
152  m_alive[m_N] = true;
153  m_N++;
154  }
155  }
156  // kill the walker
157  if (survivors == 0)
158  m_alive[n] = false;
159  // too many walkers ...
160  if (m_N > 500000) {
161  cout << "Too many nodes: " << m_N << std::endl;
162  exit(-1);
163  }
164 }

166 void SchroedingerDiffusionMC::MC() {
167  int N0 = m_N; // original number of walkers
168  for (int i : irange<int>(0, N0)) step(i); // make N0 MCMC steps
169 
170  // now count the number of alive walkers
171  // this also cleans up the vectors removing the dead walkers from the list
172  int j = 0;
173  for (int i : irange<int>(0, m_N)) {
174  if (m_alive[i] && i != j) {
175  m_x[j] = m_x[i];
176  m_alive[j] = m_alive[i];
177  }
178  if (m_alive[i]) ++j;
179  }
180  m_N = j; // reset number of alive walkers
181  // https://www.thphys.uni-heidelberg.de/~wetzel/qmc2006/KOSZ96.pdf
182  // page 3
183  // the energy must be set to go in the direction of the
184  // ground state energy, otherwise the wave function vanishes or diverges
185  // see end of page 5, beginning of page 6
186  // change in number of walkers implies change in energy
187  // setting alpha = 0.1
188  m_E0 += log(double(m_NT)/double(m_N))/10.0;
189  // calculate average energy and average energy^2
190  m_sumE += m_E0;
191  m_sumE2 += m_E0*m_E0;
192  // finally, histogram in this step for each walker
193  for (int i : irange<int>(0, m_N)) {
194  // find out the position of each walker
195  double x = m_x[i][0];
196  // find the index of the walker in psi
197  int k = int((x-m_xmin)/(m_xmax-m_xmin)*m_psi.size());
198  // increment psi in that index to histogram it
199  if (k < m_psi.size() && k > 0)
200  m_psi[k] += 1;
201  }
202  m_nMCSteps++; // end of this MCMC step
203 }
