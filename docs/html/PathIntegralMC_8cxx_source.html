114 bool PathIntegralMC::step(double &x_n) {
115  uniform_int_distribution<int> rFlatInt(0, m_Nt-1);
116  // choose random time slice:
117  int t = rFlatInt(rEngine);
118 
119  int t_prev = std::max(0, t-1);
120  int t_next = std::min(m_Nt-1, t+1);
121 
122  uniform_real_distribution<double> rFlat(-1, 1);
123  // shift walker in time t
124  // P = exp( - m (x_n - x_{n-1})^2/dt ) incorporates the kinetic energy
125  // and it is simulated by the random displacement below
126  x_n = m_x[t] + m_delta*rFlat(rEngine); // shifts walker uniformly
127  // the equation above is the conditional probability of moving from one
128  // step to the next: G(x'|x) = G(x|x') = x + delta*flat(-1,1)
129 
130  // now check if we should accept it
131  // to generate paths with probability distribution given by
132  // the density function, we need to use it in the balance probability equation
133  // trial is accepted with probability A
134  // A(x'|x) = min (1, P(x') g(x|x') / [P(x) g(x'|x)] )
135  double deltaE = 0; // change in the energy
136  deltaE += V(x_n) - V(m_x[t]); // change in potential energy
137  deltaE += 0.5*std::pow((m_x[t_next] - x_n)/m_dt, 2) - 0.5*std::pow((m_x[t_next] - m_x[t])/m_dt, 2); // change in kinetic energy
138  deltaE += 0.5*std::pow((x_n - m_x[t_prev])/m_dt, 2) - 0.5*std::pow((m_x[t] - m_x[t_prev])/m_dt, 2); // change in kinetic energy
139  double A = std::exp(-m_dt*deltaE);
140  if (A > 1) A = 1;
141  uniform_real_distribution<double> rFlatA(0, 1);
142  if (A > rFlatA(rEngine)) { // accept
143  m_x[t] = x_n;
144  return true;
145  }
146  return false;
147 }

149 void PathIntegralMC::MC() {
150  for (int i : irange<int>(0, m_Nt)) {
151  double x_n = 0;
152  bool accepted = step(x_n); // make Nt MCMC steps
153 
154  if (!accepted) continue;
155 
156  double E0 = V(x_n) + 0.5*x_n*dVdx(x_n); // kinetic energy from the Virial theorem
157  // calculate average energy and average energy^2
158  m_sumE += E0;
159  m_sumE2 += E0*E0;
160 
161  // find the index of the walker in psi
162  int k = int((x_n-m_xmin)/(m_xmax-m_xmin)*m_psi.size());
163  // increment psi in that index to histogram it
164  if (k < m_psi.size() && k >= 0)
165  m_psi[k] += 1;
166  }
167  m_nMCSteps++; // end of this MCMC step
168 }
